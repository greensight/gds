    {({ theme: themeName, ...args }) => {
        const basicTheme = {
            button: (state) => {
                const sized = {
                    sm: {
                        ...typography('buttonSm'),
                        borderRadius: 14,
                    },
                    md: {
                        ...typography('buttonMd'),
                        lineHeight: 1,
                        borderRadius: scale(2),
                    },
                    lg: {
                        ...typography('buttonLg'),
                        borderRadius: scale(2),
                    },
                };
                const varianted = {
                    primary: {
                        color: '#fff',
                        background: 'grey',
                        ':hover': {
                            background: 'darkgrey',
                        },
                    },
                    ghost: {
                        border: '2px solid grey',
                        ':hover': {
                            background: '#fafafa',
                        },
                    },
                };
                return {
                    padding: scale(2),
                    display: 'inline-flex',
                    width: 'fit-content',
                    justifyContent: 'center',
                    alignItems: 'center',
                    ...(state.block && {
                        display: 'block',
                        width: '100%',
                    }),
                    ...(state.rounded && {
                        borderRadius: 12,
                    }),
                    ...extractCSSOption(sized, state.size),
                    ...extractCSSOption(varianted, state.variant),
                };
            },
            icon: (state) => {
                const marginRule = `margin${!state.iconAfter ? 'Right' : 'Left'}`;
                const invMarginRule = `margin${state.iconAfter ? 'Right' : 'Left'}`;
                const iconCSS = {
                    [marginRule]: !state.hidden ? (state.hasChildren ? 1 : -1) * scale(1, true) : undefined,
                    [invMarginRule]: !state.hidden ? -scale(1, true) : undefined,
                    width: scale(3, true),
                    height: scale(3, true),
                };
                return { ...iconCSS, fill: 'currentcolor' };
            },
        };
        const themes = {
            basic: basicTheme,
        };
        const Button = createButtonWithTheme();
        return (
            <Button {...args} theme={themes[themeName]} onClick={action('onClick')}>
                I'm a button
            </Button>
        );
    }}
</Story>
